### 2.3 `scraper_partidos.py` (hacer robusto para Render)

* Debe exponer como mínimo:

```python
# URL objetivo
TARGET_URL = "https://live21.nowgoal25.com/..."  # deja la que uses ahora

# 1) Requests con UA y timeout
import requests

UA = "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/120.0.0.0 Safari/537.36"

def fetch_with_requests(url: str, timeout: int = 20) -> str | None:
    try:
        r = requests.get(url, headers={"User-Agent": UA}, timeout=timeout)
        if r.status_code == 200:
            return r.text
        return None
    except Exception:
        return None

# 2) Detección de filas/odds en el HTML
def html_has_rows(html: str) -> bool:
    if not html:
        return False
    # Ajusta a tu selector real de filas con odds/handicap:
    return ("tr1_" in html) or ('odds="' in html)  # mínimo; afina si ya tienes selectores

# 3) Playwright con flags de contenedor
import asyncio
from playwright.async_api import async_playwright

PLAYWRIGHT_ARGS = ["--no-sandbox", "--disable-dev-shm-usage"]

async def fetch_with_playwright(url: str, timeout_ms: int = 20000) -> str | None:
    async with async_playwright() as p:
        browser = await p.chromium.launch(headless=True, args=PLAYWRIGHT_ARGS)
        context = await browser.new_context(user_agent=UA)
        page = await context.new_page()
        try:
            await page.goto(url, wait_until="networkidle", timeout=timeout_ms)
            # Espera al selector clave (ajústalo a tu tabla real):
            try:
                await page.wait_for_selector("tr[id^='tr1_']", state="attached", timeout=timeout_ms)
            except Exception:
                pass
            html = await page.content()
            return html
        finally:
            await context.close()
            await browser.close()

def fetch_html_via_playwright_sync(url: str, timeout_ms: int = 20000) -> str | None:
    try:
        return asyncio.run(fetch_with_playwright(url, timeout_ms=timeout_ms))
    except RuntimeError:
        loop = asyncio.new_event_loop()
        try:
            asyncio.set_event_loop(loop)
            return loop.run_until_complete(fetch_with_playwright(url, timeout_ms=timeout_ms))
        finally:
            loop.close()
```

* **Pipeline recomendado** donde ya parseas y normalizas tiempos a UTC (mantén tus funciones actuales de parseo; no cambies firmas). Si hay filtros “próximos”, que comparen en UTC.
