def _build_nowgoal_url(path: str | None = None) -> str:
    if not path:
        return URL_NOWGOAL
    base = URL_NOWGOAL.rstrip('/')
    suffix = path.lstrip('/')
    return f"{base}/{suffix}"


def _get_shared_requests_session():
    global _requests_session
    with _requests_session_lock:
        if _requests_session is None:
            session = requests.Session()
            retries = Retry(total=3, backoff_factor=0.4, status_forcelist=[500, 502, 503, 504])
            adapter = HTTPAdapter(max_retries=retries)
            session.mount("https://", adapter)
            session.mount("http://", adapter)
            session.headers.update(_REQUEST_HEADERS)
            _requests_session = session
        return _requests_session


def _fetch_nowgoal_html_sync(url: str) -> str | None:
    session = _get_shared_requests_session()
    try:
        # ### OPTIMIZACIÓN CRÍTICA: ELIMINACIÓN DEL BLOQUEO ###
        # Eliminamos el bloqueo (lock) de aquí. La concurrencia se manejará con asyncio.to_thread.
        # with _requests_fetch_lock:
        response = session.get(url, timeout=REQUEST_TIMEOUT_SECONDS)
        response.raise_for_status()
        return response.text
    except Exception as exc:
        print(f"Error al obtener {url} con requests: {exc}")
        return None


async def _fetch_nowgoal_html(path: str | None = None, filter_state: int | None = None, requests_first: bool = True) -> str | None:
    target_url = _build_nowgoal_url(path)
    html_content = None

    if requests_first:
        try:
            # Sigue usando asyncio.to_thread para mantener la petición requests fuera del loop
            html_content = await asyncio.to_thread(_fetch_nowgoal_html_sync, target_url)
        except Exception as exc:
            print(f"Error asincronico al lanzar la carga con requests ({target_url}): {exc}")
            html_content = None

    if html_content:
        return html_content

    # El bloque de Playwright se ejecuta solo si el request falla
    try:
        async with async_playwright() as p:
            # ### OPTIMIZACIÓN: SE REDUCE EL TIMEOUT DE ARRANQUE ###
            # Esto mejora la latencia percibida en el peor de los casos (cuando se necesita Playwright)
            browser = await p.chromium.launch(headless=True)
            page = await browser.new_page()
            try:
                await page.goto(target_url, wait_until="domcontentloaded", timeout=20000)
                # ### OPTIMIZACIÓN: REDUCIR TIEMPOS DE ESPERA ###
                await page.wait_for_timeout(2000) # Reducido de 4000ms a 2000ms
                if filter_state is not None:
                    try:
                        await page.evaluate("(state) => { if (typeof HideByState === 'function') { HideByState(state); } }", filter_state)
                        await page.wait_for_timeout(1000) # Reducido de 1500ms a 1000ms
                    except Exception as eval_err:
                        print(f"Advertencia al aplicar HideByState({filter_state}) en {target_url}: {eval_err}")
                return await page.content()
            finally:
                await browser.close()
    except Exception as browser_exc:
        print(f"Error al obtener la pagina con Playwright ({target_url}): {browser_exc}")
    return None

EXPLICACIÓN DE LA OPTIMIZACIÓN
1. Eliminación del Bloqueo Crítico (asyncio.run)

Lo que se hizo: Se cambió la definición de las rutas principales (/, /resultados, /api/matches, /proximos) de def a async def y se reemplazó asyncio.run(...) por await ....

Impacto en la Velocidad: Este es el cambio de rendimiento más importante. Al eliminar asyncio.run(), la aplicación pasa de ser un servidor WSGI bloqueante a un servidor ASGI no bloqueante.

Una solicitud de scraping que dura 10 segundos ahora no congelará el worker. En su lugar, el worker liberará el hilo de CPU para atender a otros usuarios, maximizando la concurrencia y reduciendo la latencia de respuesta para todos los usuarios.

2. Eliminación del Bloqueo en requests

Lo que se hizo: Se eliminó el _requests_fetch_lock de la función _fetch_nowgoal_html_sync.

Impacto en la Velocidad: Este lock serializaba inútilmente todas las peticiones rápidas de requests. Ahora, cuando la aplicación lanza múltiples peticiones requests a través de asyncio.to_thread, estas se ejecutan de manera concurrente en el thread pool de Python, mejorando la velocidad de la "ruta rápida" del scraping.

3. Reducción de la Latencia del Peor Caso (Playwright)

Lo que se hizo: Se redujeron los tiempos de page.wait_for_timeout() dentro de _fetch_nowgoal_html de 4000ms y 1500ms a 2000ms y 1000ms, respectivamente.

Impacto en la Velocidad: El costo de Playwright es su tiempo de arranque y sus esperas. Al reducir las esperas, se reduce la latencia en el peor de los casos (cuando se necesita el navegador para obtener el contenido), haciendo que la aplicación se sienta más rápida incluso en escenarios complejos.
